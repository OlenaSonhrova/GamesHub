переходимо в цю папку від імені адміністратора: C:\Program Files\Java\jdkx.x.x_x\bin 
вводимо цю команду в терміналі: keytool -genkeypair -v -storetype PKCS12 -keystore my-upload-key.keystore -alias my-key-alias -keyalg RSA -keysize 2048 -validity 10000
Розмістіть my-upload-key.keystoreфайл у android/appкаталозі папки проекту.
Відредагуйте файл ~/.gradle/gradle.propertiesабо android/gradle.propertiesта додайте наступне (замініть *****правильним паролем сховища ключів, псевдонімом і паролем ключа),
MYAPP_UPLOAD_STORE_FILE=my-upload-key.keystore
MYAPP_UPLOAD_KEY_ALIAS=my-key-alias
MYAPP_UPLOAD_STORE_PASSWORD=*****
MYAPP_UPLOAD_KEY_PASSWORD=*****

import { useEffect, useState } from 'react';
import { FlatList, Pressable, SafeAreaView, StyleSheet, Text, View, Image, RefreshControl } from 'react-native';
import NetInfo from "@react-native-community/netinfo";
import AsyncStorage from '@react-native-async-storage/async-storage';
import { useQuery, QueryClient } from '@tanstack/react-query';
import Loader from '../../registration/components/loader';
import { GetAllGameTypes } from '../../api/api';

const queryClient = new QueryClient();

const Games = ({ navigation }) => {


	const [localData, setLocalData] = useState(null);

	useEffect(() => {
		const fetchLocalData = async () => {
			const storedData = await AsyncStorage.getItem('gameTypes');
			if (storedData) {
				setLocalData(JSON.parse(storedData));
			}
		};
		fetchLocalData();
	}, []);

	const { data, isLoading, isError, error, refetch } = useQuery(

		{
			queryKey: ["getAllGameTypes"],
			queryFn: GetAllGameTypes,
			staleTime: 3600000,
			initialData: localData,
		}
	);

	const handleRefresh = async () => {
		const response = await refetch();
		if (response?.data == 'AxiosError: Network Error') {
			queryClient.setQueryData(["getAllGameTypes"], localData);
		};
	};


	if (isLoading) {
		return <Loader />;
	};


	if (isError) {
		alert('return. isError', error);
		return null;
	};

	return (
		<View style={styles.backgroundColor}>
			<SafeAreaView style={styles.container}>
				<Text style={styles.titleBlock}>КАТЕГОРІЇ</Text>
				<FlatList
					data={localData}
					refreshControl={
						<RefreshControl
							refreshing={isLoading}
							onRefresh={handleRefresh}
						/>
					}
					renderItem={({ item, index }) => (
						<Pressable style={({ pressed }) => [
							{
								backgroundColor: pressed ? '#FAE2D4' : '#EEC9B0',
							},
							styles.block,
						]} onPress={() => {
							navigation.navigate('Ігри', {
								type: item.name,
							});
						}}>
							<View style={styles.blockText}>
								<Text style={styles.number}>{item.count}</Text>
								<Text style={styles.text}>{item.name}</Text>
							</View>
							{/* <Image style={styles.image} key={index} source={item.src} /> */}
						</Pressable>
					)}
				/>
			</SafeAreaView>
		</View>
	);
};
